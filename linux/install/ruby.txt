cd ~/source

setenv RUBY_VER 1.8.7
setenv PATCH_LEV 160

wget ftp://ftp.ruby-lang.org/pub/ruby/1.8/ruby-1.8.7-p160.tar.gz
tar xzf ruby-${RUBY_VER}-p${PATCH_LEV}.tar.gz 

cd ruby-${RUBY_VER}-p${PATCH_LEV}

setenv RUBY_HOME /usr/local/ruby-${RUBY_VER}-p${PATCH_LEV}

./configure --prefix=${RUBY_HOME}
make

sudo make install

set path=(${RUBY_HOME}/bin $path)

cd ext/zlib
ruby extconf.rb
make
sudo make install


cd /usr/local
sudo rm ruby
sudo ln -s ruby-${RUBY_VER}-p${PATCH_LEV} ruby
